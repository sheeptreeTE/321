    <div id="blessing-box"></div>

    <script>
        function startBomber() {
            let count = parseInt(document.getElementById('countInput').value);
            let box = document.getElementById('blessing-box');
            box.innerHTML = ""; 
            let i = 0;
            let interval = setInterval(() => {
                if (i < count) {
                    box.innerHTML += "✨ Happy ！！！ ✨<br>";
                    box.scrollTop = box.scrollHeight; 
                    i++;
                } else {
                    clearInterval(interval);
                    box.innerHTML += "<br>🎊 轰炸结束！祝你天天开心！";
                }
            }, 50); 
        }
    </script>
